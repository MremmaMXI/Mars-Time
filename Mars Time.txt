function dateToJD(d){
  return d.getTime()/86400000 + 2440587.5;
}

function getMSD(date){
  const jdTT = dateToJD(date) + 69.184/86400;
  return (jdTT - MSD_EPOCH_JD) / MSD_DIVISOR;
}

function getMTC(msd){
  return (msd - Math.floor(msd)) * 24;
}

function mtcToHMS(mtcH){
  const tot = Math.floor(mtcH * 3600);
  return {h:Math.floor(tot/3600), m:Math.floor((tot%3600)/60), s:tot%60};
}

// Soles (con fracción) desde la época Silin hasta la fecha dada
function solsFromEpoch(date){
  return (date.getTime() - SILIN_EPOCH_MS) / 86400000 / MARS_SOL_RATIO;
}

// Fecha JS desde soles desde época Silin
function dateFromSols(totalSols){
  return new Date(SILIN_EPOCH_MS + totalSols * MARS_SOL_RATIO * 86400000);
}

// Longitud del mes 12 de un año marciano
function etimberLen(marsYear){
  const startSol = Math.round(MARS_YEAR_SOLS * (marsYear - 1));
  const endSol   = Math.round(MARS_YEAR_SOLS * marsYear);
  return endSol - startSol - 616; // 52, 53 o 54
}

// Soles acumulados al inicio de un año marciano (desde época)
function solsAtYearStart(marsYear){
  return Math.round(MARS_YEAR_SOLS * (marsYear - 1));
}

// Soles al inicio del mes m del año y (0-indexed dentro del año)
function solsAtMonthStart(marsYear, month){
  let s = solsAtYearStart(marsYear);
  for(let i=1;i<month;i++) s += (i<=11)?56:etimberLen(marsYear);
  return s;
}

// Longitud del mes m del año y
function marsMonthLen(marsYear, month){
  return (month<=11)?56:etimberLen(marsYear);
}

// Convierte soles totales → fecha Silin
function silinFromTotalSols(totalSolsFloat){
  const totalSols = Math.floor(totalSolsFloat);
  const fracSol   = totalSolsFloat - totalSols;

  // Buscar el año
  let year = Math.max(1, Math.floor(totalSols / MARS_YEAR_SOLS) + 1);
  // Ajuste fino
  while(year > 1 && solsAtYearStart(year) > totalSols) year--;
  while(solsAtYearStart(year+1) <= totalSols) year++;

  const solInYear = totalSols - solsAtYearStart(year); // 0-indexed
  let month = 1, rem = solInYear;
  for(let m=1;m<=12;m++){
    const ml = marsMonthLen(year, m);
    if(rem < ml){ month = m; break; }
    rem -= ml;
  }

  const weekday = solInYear % 7; // 0=Lúnares

  return {year, month, sol:rem+1, weekday, solOfYear:solInYear+1, fracSol};
}

// Total soles desde época para una fecha Silin exacta
function totalSolsForSilin(marsYear, month, sol){
  return solsAtMonthStart(marsYear, month) + (sol - 1);
}

function getJDN(y,m,d){
  return Math.floor((1461*(y+4800+Math.floor((m-14)/12)))/4)
       + Math.floor((367*(m-2-12*Math.floor((m-14)/12)))/12)
       - Math.floor((3*Math.floor((y+4900+Math.floor((m-14)/12))/100))/4)
       + d - 32075;
}

function getDOY(y,m,d){
  return Math.round((Date.UTC(y,m-1,d)-Date.UTC(y,0,1))/86400000)+1;
}

function getISOWeek(date){
  const d=new Date(Date.UTC(date.getUTCFullYear(),date.getUTCMonth(),date.getUTCDate()));
  const day=d.getUTCDay()||7; d.setUTCDate(d.getUTCDate()+4-day);
  const ys=new Date(Date.UTC(d.getUTCFullYear(),0,1));
  return Math.ceil(((d-ys)/86400000+1)/7);
}

function daysInMonth(y,m){ return new Date(Date.UTC(y,m,0)).getUTCDate(); }

// Fecha terrestre → fecha Silin
function earthToSilin(y,m,d){
  const date = new Date(Date.UTC(y,m-1,d));
  const sols  = solsFromEpoch(date);
  return silinFromTotalSols(sols);
}

// Fecha Silin → fecha terrestre (objeto Date UTC)
function silinToEarth(marsYear, month, sol){
  const ts = totalSolsForSilin(marsYear, month, sol);
  return dateFromSols(ts);
}

    ctx.beginPath(); ctx.moveTo(cx,cy);
    ctx.lineTo(cx+Math.cos(angle)*len, cy+Math.sin(angle)*len);
    ctx.strokeStyle=col; ctx.lineWidth=w; ctx.lineCap='round'; ctx.stroke(); ctx.restore();
  }

}

  for(let i=0;i<firstDow;i++){ html+='<td class="empty"></td>'; col++; }
  for(let d=1;d<=totalDays;d++){
    if(col>0&&col%7===0) html+='</tr><tr>';
    const isToday = (navYear===todayYear && navMonth===todayMonth && d===todayDay);
    const isSel   = (selDay===d);
    let cls = '';
    if(isToday&&isSel) cls='today selected';
    else if(isToday)   cls='today';
    else if(isSel)     cls='selected';
    html+=`<td class="${cls}" onclick="onEarthClick(${d})">${d}</td>`;
    col++;
  }
  while(col%7!==0){ html+='<td class="empty"></td>'; col++; }
  html+='</tr></tbody>';
  document.getElementById('cal-earth-table').innerHTML=html;

  document.getElementById('earth-cal-title').textContent=`${EARTH_MONTHS[navMonth-1]} ${navYear}`;
  const weeks = Math.ceil((firstDow+totalDays)/7);
  document.getElementById('earth-cal-sub').textContent=`${totalDays} días · ${weeks} semanas`;

  // Botones límite
  document.getElementById('e-yy-prev').disabled = (navYear<=1);
  document.getElementById('e-mm-prev').disabled = (navYear<=1 && navMonth<=1);
  document.getElementById('e-mm-next').disabled = (navYear>=MAX_EARTH_YEAR && navMonth>=12);
  document.getElementById('e-yy-next').disabled = (navYear>=MAX_EARTH_YEAR);
}

/* ══════════════════════════════════════════════
   CONSTRUIR CALENDARIO MARCIANO
══════════════════════════════════════════════ */
function buildMarsCal(navYear, navMonth, todayYear, todayMonth, todaySol, selSol){
  const monthLen  = marsMonthLen(navYear, navMonth);
  // El Sol 1 del año es siempre Lúnares (0)
  // Calculamos el índice del Sol 1 de este mes dentro del año
  let solsBeforeThisMonth = 0;
  for(let m=1;m<navMonth;m++) solsBeforeThisMonth += marsMonthLen(navYear, m);
  const startWeekday = solsBeforeThisMonth % 7; // 0=Lúnares

  let html='<thead><tr>';
  MARS_DAYS_SHORT.forEach(d=>{ html+=`<th>${d}</th>`; });
  html+='</tr></thead><tbody><tr>';

  let col=0;
  for(let i=0;i<startWeekday;i++){ html+='<td class="empty"></td>'; col++; }
  for(let s=1;s<=monthLen;s++){
    if(col>0&&col%7===0) html+='</tr><tr>';
    const isToday = (navYear===todayYear && navMonth===todayMonth && s===todaySol);
    const isSel   = (selSol===s);
    let cls='';
    if(isToday&&isSel) cls='today selected';
    else if(isToday)   cls='today';
    else if(isSel)     cls='selected';
    html+=`<td class="${cls}" onclick="onMarsClick(${s})">${s}</td>`;
    col++;
  }
  while(col%7!==0){ html+='<td class="empty"></td>'; col++; }
  html+='</tr></tbody>';
  document.getElementById('cal-mars-table').innerHTML=html;

  document.getElementById('mars-cal-title').textContent=`${MARS_MONTHS[navMonth-1]} · MY ${navYear}`;
  const weeks=Math.ceil((startWeekday+monthLen)/7);
  document.getElementById('mars-cal-sub').textContent=`${monthLen} soles · ${weeks} semanas`;

  document.getElementById('m-yy-prev').disabled=(navYear<=1);
  document.getElementById('m-mm-prev').disabled=(navYear<=1&&navMonth<=1);
  document.getElementById('m-mm-next').disabled=(navYear>=MAX_MARS_YEAR&&navMonth>=12);
  document.getElementById('m-yy-next').disabled=(navYear>=MAX_MARS_YEAR);
}

/* ══════════════════════════════════════════════
   NAVEGAR TIERRA
══════════════════════════════════════════════ */
function earthNav(unit, dir){
  if(unit==='mm'){
    eNavMonth += dir;
    if(eNavMonth>12){ eNavMonth=1; eNavYear++; }
    if(eNavMonth<1){ eNavMonth=12; eNavYear--; }
  } else {
    eNavYear += dir;
  }
  eNavYear = Math.max(1, Math.min(MAX_EARTH_YEAR, eNavYear));
  eNavMonth = Math.max(1, Math.min(12, eNavMonth));
  eSelDay = null;
  eNavLive = false;
  updateEarthSelInfo(null);
  renderEarth(new Date());

  if(syncMode){
    // Sincronizar: ir al día 1 del mes navegado en Tierra
    const d = new Date(Date.UTC(eNavYear, eNavMonth-1, 1));
    const s = earthToSilin(eNavYear, eNavMonth, 1);
    mNavYear  = s.year;
    mNavMonth = s.month;
    mSelSol   = null;
    mNavLive  = false;
    renderMars(new Date());
  }
}

/* ══════════════════════════════════════════════
   NAVEGAR MARTE
══════════════════════════════════════════════ */
function marsNav(unit, dir){
  if(unit==='mm'){
    mNavMonth += dir;
    if(mNavMonth>12){ mNavMonth=1; mNavYear++; }
    if(mNavMonth<1){ mNavMonth=12; mNavYear--; }
  } else {
    mNavYear += dir;
  }
  mNavYear  = Math.max(1, Math.min(MAX_MARS_YEAR, mNavYear));
  mNavMonth = Math.max(1, Math.min(12, mNavMonth));
  mSelSol   = null;
  mNavLive  = false;
  updateMarsSelInfo(null);
  renderMars(new Date());

  if(syncMode){
    const earthDate = silinToEarth(mNavYear, mNavMonth, 1);
    eNavYear  = earthDate.getUTCFullYear();
    eNavMonth = earthDate.getUTCMonth()+1;
    eSelDay   = null;
    eNavLive  = false;
    renderEarth(new Date());
  }
}

/* ══════════════════════════════════════════════
   CLICK EN DÍA
══════════════════════════════════════════════ */
function onEarthClick(d){
  eSelDay  = (eSelDay===d) ? null : d; // toggle
  eNavLive = false;
  const now = new Date();
  renderEarth(now);

  const selDateStr = eSelDay
    ? `${EARTH_DAYS_FULL[new Date(Date.UTC(eNavYear,eNavMonth-1,eSelDay)).getUTCDay()]}, ${eSelDay} de ${EARTH_MONTHS[eNavMonth-1]} de ${eNavYear}`
    : null;
  updateEarthSelInfo(selDateStr);

  if(syncMode && eSelDay){
    const s = earthToSilin(eNavYear, eNavMonth, eSelDay);
    mNavYear  = s.year;
    mNavMonth = s.month;
    mSelSol   = s.sol;
    mNavLive  = false;
    renderMars(now);
    updateMarsSelInfo(`${MARS_DAYS_FULL[s.weekday]}, Sol ${s.sol} de ${MARS_MONTHS[s.month-1]}, MY ${s.year}`);
  }
}

function onMarsClick(sol){
  mSelSol  = (mSelSol===sol) ? null : sol;
  mNavLive = false;
  const now = new Date();
  renderMars(now);

  const weekdayIdx = (solsAtMonthStart(mNavYear,mNavMonth) + (mSelSol||1) - 1) % 7;
  const selStr = mSelSol
    ? `${MARS_DAYS_FULL[weekdayIdx]}, Sol ${mSelSol} de ${MARS_MONTHS[mNavMonth-1]}, MY ${mNavYear}`
    : null;
  updateMarsSelInfo(selStr);

  if(syncMode && mSelSol){
    const ed = silinToEarth(mNavYear, mNavMonth, mSelSol);
    eNavYear  = ed.getUTCFullYear();
    eNavMonth = ed.getUTCMonth()+1;
    eSelDay   = ed.getUTCDate();
    eNavLive  = false;
    renderEarth(now);
    updateEarthSelInfo(`${EARTH_DAYS_FULL[ed.getUTCDay()]}, ${eSelDay} de ${EARTH_MONTHS[eNavMonth-1]} de ${eNavYear}`);
  }
}

/* ══════════════════════════════════════════════
   RENDER TIERRA
══════════════════════════════════════════════ */
function renderEarth(now){
  const tY=now.getUTCFullYear(), tM=now.getUTCMonth()+1, tD=now.getUTCDate();

  // Fecha mostrada en el panel principal = fecha actual real (no nav)
  document.getElementById('earth-day').textContent        = tD;
  document.getElementById('earth-month-year').textContent = `${EARTH_MONTHS[tM-1]} ${tY}`;
  document.getElementById('earth-weekday').textContent    = EARTH_DAYS_FULL[now.getUTCDay()];

  // Info row (siempre fecha real)
  document.getElementById('earth-jdn').textContent  = getJDN(tY,tM,tD);
  document.getElementById('earth-doy').textContent  = getDOY(tY,tM,tD);
  document.getElementById('earth-week').textContent = `S${pad(getISOWeek(now))}`;

  // Botón Hoy
  document.getElementById('earth-now-btn').classList.toggle('away', !eNavLive);

  // Calendario con mes de navegación
  buildEarthCal(eNavYear, eNavMonth, tY, tM, tD, eSelDay);
}

/* ══════════════════════════════════════════════
   RENDER MARTE
══════════════════════════════════════════════ */
function renderMars(now){
  const msd     = getMSD(now);
  const mtcH    = getMTC(msd);
  const {h,m,s} = mtcToHMS(mtcH);

  const totalSols = solsFromEpoch(now);
  const silin     = silinFromTotalSols(totalSols);

  // Panel principal = fecha marciana real ahora
  document.getElementById('mars-day').textContent        = silin.sol;
  document.getElementById('mars-month-year').textContent = `${MARS_MONTHS[silin.month-1]} MY ${silin.year}`;
  document.getElementById('mars-weekday').textContent    = MARS_DAYS_FULL[silin.weekday];

  document.getElementById('mars-msd').textContent = msd.toFixed(2);
  document.getElementById('mars-soy').textContent = silin.solOfYear;
  document.getElementById('mars-my').textContent  = silin.year;

  // Botón Ahora
  document.getElementById('mars-now-btn').classList.toggle('away', !mNavLive);

  // Calendario con mes de navegación
  buildMarsCal(mNavYear, mNavMonth, silin.year, silin.month, silin.sol, mSelSol);
}

/* ══════════════════════════════════════════════
   TICK PRINCIPAL
══════════════════════════════════════════════ */
function tick(){
  const now = new Date();

  // ── Tierra (reloj siempre en tiempo real) ──
  const eH=now.getUTCHours(), eMi=now.getUTCMinutes(), eSe=now.getUTCSeconds();
  document.getElementById('digital-earth').textContent=`${pad(eH)}:${pad(eMi)}:${pad(eSe)}`;
  drawClock('canvas-earth', eH, eMi, eSe, EP);

  // Actualizar nav si está en live
  if(eNavLive){
    eNavYear=now.getUTCFullYear(); eNavMonth=now.getUTCMonth()+1;
  }
  renderEarth(now);

  // ── Marte (reloj siempre en tiempo real) ──
  const msd     = getMSD(now);
  const mtcH    = getMTC(msd);
  const {h,m,s} = mtcToHMS(mtcH);
  document.getElementById('digital-mars').textContent=`${pad(h)}:${pad(m)}:${pad(s)}`;
  drawClock('canvas-mars', h, m, s, MP);

  if(mNavLive){
    const silin=silinFromTotalSols(solsFromEpoch(now));
    mNavYear=silin.year; mNavMonth=silin.month;
  }
  renderMars(now);
}
